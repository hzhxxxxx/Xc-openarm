**src/realsense_node.cpp:**
```cpp
#include "openarm_sensors/realsense_node.hpp"
#include <cv_bridge/cv_bridge.h>
#include <opencv2/opencv.hpp>

namespace openarm_sensors
{

RealSenseNode::RealSenseNode(const rclcpp::NodeOptions & options)
: Node("realsense_node", options)
{
  // 声明参数
  this->declare_parameter("width", 640);
  this->declare_parameter("height", 480);
  this->declare_parameter("fps", 30);
  this->declare_parameter("frame_id", "camera_link");

  // 获取参数
  width_ = this->get_parameter("width").as_int();
  height_ = this->get_parameter("height").as_int();
  fps_ = this->get_parameter("fps").as_int();
  frame_id_ = this->get_parameter("frame_id").as_string();

  // 配置RealSense
  cfg_.enable_stream(RS2_STREAM_COLOR, width_, height_, RS2_FORMAT_BGR8, fps_);
  cfg_.enable_stream(RS2_STREAM_DEPTH, width_, height_, RS2_FORMAT_Z16, fps_);

  // 启动管道
  try {
    pipe_.start(cfg_);
    RCLCPP_INFO(this->get_logger(), "RealSense pipeline started");
  } catch (const rs2::error & e) {
    RCLCPP_ERROR(this->get_logger(), "Failed to start RealSense: %s", e.what());
    throw;
  }

  // 创建发布器
  color_pub_ = this->create_publisher<sensor_msgs::msg::Image>(
    "~/color/image_raw", 10);
  depth_pub_ = this->create_publisher<sensor_msgs::msg::Image>(
    "~/depth/image_raw", 10);
  info_pub_ = this->create_publisher<sensor_msgs::msg::CameraInfo>(
    "~/camera_info", 10);

  // 定时器
  timer_ = this->create_wall_timer(
    std::chrono::milliseconds(1000 / fps_),
    std::bind(&RealSenseNode::timer_callback, this));

  RCLCPP_INFO(this->get_logger(), "RealSense node initialized");
}

RealSenseNode::~RealSenseNode()
{
  pipe_.stop();
}

void RealSenseNode::timer_callback()
{
  publish_images();
}

void RealSenseNode::publish_images()
{
  // 等待帧
  rs2::frameset frames = pipe_.wait_for_frames();

  // 彩色图像
  rs2::video_frame color_frame = frames.get_color_frame();
  cv::Mat color_mat(
    cv::Size(width_, height_),
    CV_8UC3,
    (void*)color_frame.get_data(),
    cv::Mat::AUTO_STEP
  );

  auto color_msg = cv_bridge::CvImage(
    std_msgs::msg::Header(), "bgr8", color_mat).toImageMsg();
  color_msg->header.stamp = this->now();
  color_msg->header.frame_id = frame_id_;
  color_pub_->publish(*color_msg);

  // 深度图像
  rs2::depth_frame depth_frame = frames.get_depth_frame();
  cv::Mat depth_mat(
    cv::Size(width_, height_),
    CV_16UC1,
    (void*)depth_frame.get_data(),
    cv::Mat::AUTO_STEP
  );

  auto depth_msg = cv_bridge::CvImage(
    std_msgs::msg::Header(), "16UC1", depth_mat).toImageMsg();
  depth_msg->header.stamp = this->now();
  depth_msg->header.frame_id = frame_id_;
  depth_pub_->publish(*depth_msg);

  // 相机信息
  auto info_msg = std::make_shared<sensor_msgs::msg::CameraInfo>();
  info_msg->header.stamp = this->now();
  info_msg->header.frame_id = frame_id_;
  info_msg->width = width_;
  info_msg->height = height_;
  // 填充内参矩阵...
  info_pub_->publish(*info_msg);
}

} // namespace openarm_sensors

#include <rclcpp_components/register_node_macro.hpp>
RCLCPP_COMPONENTS_REGISTER_NODE(openarm_sensors::RealSenseNode)
```

## 3. 集成末端执行器

### 3.1 场景: 添加Robotiq 2F-85夹爪

#### 步骤1: 创建夹爪硬件接口

**在openarmx_hardware中添加 gripper_interface.hpp:**
```cpp
#ifndef openarmx_hardware__GRIPPER_INTERFACE_HPP_
#define openarmx_hardware__GRIPPER_INTERFACE_HPP_

#include <hardware_interface/system_interface.hpp>
#include <rclcpp/rclcpp.hpp>
#include <modbus/modbus.h>

namespace openarmx_hardware
{

class RobotiqGripperInterface : public hardware_interface::SystemInterface
{
public:
  CallbackReturn on_init(const hardware_interface::HardwareInfo & info) override;
  CallbackReturn on_configure(const rclcpp_lifecycle::State & previous_state) override;
  CallbackReturn on_activate(const rclcpp_lifecycle::State & previous_state) override;
  CallbackReturn on_deactivate(const rclcpp_lifecycle::State & previous_state) override;

  hardware_interface::return_type read(
    const rclcpp::Time & time, const rclcpp::Duration & period) override;
  hardware_interface::return_type write(
    const rclcpp::Time & time, const rclcpp::Duration & period) override;

  std::vector<hardware_interface::StateInterface> export_state_interfaces() override;
  std::vector<hardware_interface::CommandInterface> export_command_interfaces() override;

private:
  modbus_t * modbus_ctx_;
  std::string ip_address_;
  int port_;

  // 状态
  double position_;       // 0.0 (打开) - 0.085 (闭合) [m]
  double velocity_;
  double effort_;

  // 命令
  double position_command_;
  double velocity_command_;
  double effort_command_;

  bool activate_gripper();
  bool set_position(uint8_t target);
  uint8_t get_position();
};

} // namespace openarmx_hardware

#endif
```

**gripper_interface.cpp实现:**
```cpp
#include "openarmx_hardware/gripper_interface.hpp"
#include <cmath>

namespace openarmx_hardware
{

CallbackReturn RobotiqGripperInterface::on_init(
  const hardware_interface::HardwareInfo & info)
{
  if (hardware_interface::SystemInterface::on_init(info) != CallbackReturn::SUCCESS) {
    return CallbackReturn::ERROR;
  }

  // 读取参数
  ip_address_ = info_.hardware_parameters["ip_address"];
  port_ = std::stoi(info_.hardware_parameters["port"]);

  // 初始化状态
  position_ = 0.085;  // 初始闭合
  velocity_ = 0.0;
  effort_ = 0.0;

  position_command_ = position_;
  velocity_command_ = 0.1;     // 默认速度
  effort_command_ = 50.0;      // 默认力

  return CallbackReturn::SUCCESS;
}

CallbackReturn RobotiqGripperInterface::on_configure(
  const rclcpp_lifecycle::State & /*previous_state*/)
{
  // 创建Modbus TCP连接
  modbus_ctx_ = modbus_new_tcp(ip_address_.c_str(), port_);
  if (modbus_ctx_ == nullptr) {
    RCLCPP_ERROR(rclcpp::get_logger("RobotiqGripperInterface"),
                 "Failed to create Modbus context");
    return CallbackReturn::ERROR;
  }

  if (modbus_connect(modbus_ctx_) == -1) {
    RCLCPP_ERROR(rclcpp::get_logger("RobotiqGripperInterface"),
                 "Failed to connect to gripper at %s:%d", ip_address_.c_str(), port_);
    modbus_free(modbus_ctx_);
    return CallbackReturn::ERROR;
  }

  RCLCPP_INFO(rclcpp::get_logger("RobotiqGripperInterface"),
              "Connected to Robotiq gripper");

  return CallbackReturn::SUCCESS;
}

CallbackReturn RobotiqGripperInterface::on_activate(
  const rclcpp_lifecycle::State & /*previous_state*/)
{
  // 激活夹爪
  if (!activate_gripper()) {
    return CallbackReturn::ERROR;
  }

  RCLCPP_INFO(rclcpp::get_logger("RobotiqGripperInterface"), "Gripper activated");
  return CallbackReturn::SUCCESS;
}

CallbackReturn RobotiqGripperInterface::on_deactivate(
  const rclcpp_lifecycle::State & /*previous_state*/)
{
  modbus_close(modbus_ctx_);
  modbus_free(modbus_ctx_);
  return CallbackReturn::SUCCESS;
}

hardware_interface::return_type RobotiqGripperInterface::read(
  const rclcpp::Time & /*time*/, const rclcpp::Duration & /*period*/)
{
  // 读取夹爪位置
  uint8_t raw_position = get_position();

  // 转换为米 (0-255 -> 0.0-0.085m)
  position_ = (raw_position / 255.0) * 0.085;

  return hardware_interface::return_type::OK;
}

hardware_interface::return_type RobotiqGripperInterface::write(
  const rclcpp::Time & /*time*/, const rclcpp::Duration & /*period*/)
{
  // 转换命令 (m -> 0-255)
  uint8_t target = static_cast<uint8_t>((position_command_ / 0.085) * 255.0);

  if (!set_position(target)) {
    return hardware_interface::return_type::ERROR;
  }

  return hardware_interface::return_type::OK;
}

std::vector<hardware_interface::StateInterface>
RobotiqGripperInterface::export_state_interfaces()
{
  std::vector<hardware_interface::StateInterface> state_interfaces;

  state_interfaces.emplace_back(
    hardware_interface::StateInterface(
      "gripper_joint", hardware_interface::HW_IF_POSITION, &position_));

  return state_interfaces;
}

std::vector<hardware_interface::CommandInterface>
RobotiqGripperInterface::export_command_interfaces()
{
  std::vector<hardware_interface::CommandInterface> command_interfaces;

  command_interfaces.emplace_back(
    hardware_interface::CommandInterface(
      "gripper_joint", hardware_interface::HW_IF_POSITION, &position_command_));

  return command_interfaces;
}

bool RobotiqGripperInterface::activate_gripper()
{
  // Robotiq激活序列
  uint16_t activate_cmd = 0x0100;
  if (modbus_write_register(modbus_ctx_, 0, activate_cmd) == -1) {
    return false;
  }

  // 等待激活完成
  std::this_thread::sleep_for(std::chrono::seconds(2));
  return true;
}

bool RobotiqGripperInterface::set_position(uint8_t target)
{
  // 写入目标位置寄存器
  if (modbus_write_register(modbus_ctx_, 3, target) == -1) {
    return false;
  }
  return true;
}

uint8_t RobotiqGripperInterface::get_position()
{
  uint16_t reg[1];
  if (modbus_read_registers(modbus_ctx_, 7, 1, reg) == -1) {
    return 0;
  }
  return static_cast<uint8_t>(reg[0]);
}

} // namespace openarmx_hardware

#include <pluginlib/class_list_macros.hpp>
PLUGINLIB_EXPORT_CLASS(
  openarmx_hardware::RobotiqGripperInterface,
  hardware_interface::SystemInterface
)
```

```cpp
#ifndef OPENARM_IMPEDANCE_CONTROLLER__IMPEDANCE_CONTROLLER_HPP_
#define OPENARM_IMPEDANCE_CONTROLLER__IMPEDANCE_CONTROLLER_HPP_

#include <controller_interface/controller_interface.hpp>
#include <rclcpp/rclcpp.hpp>
#include <geometry_msgs/msg/wrench_stamped.hpp>

namespace openarm_impedance_controller
{

class ImpedanceController : public controller_interface::ControllerInterface
{
public:
  controller_interface::CallbackReturn on_init() override;

  controller_interface::InterfaceConfiguration command_interface_configuration() const override;
  controller_interface::InterfaceConfiguration state_interface_configuration() const override;

  controller_interface::CallbackReturn on_configure(
    const rclcpp_lifecycle::State & previous_state) override;
  controller_interface::CallbackReturn on_activate(
    const rclcpp_lifecycle::State & previous_state) override;
  controller_interface::CallbackReturn on_deactivate(
    const rclcpp_lifecycle::State & previous_state) override;

  controller_interface::return_type update(
    const rclcpp::Time & time, const rclcpp::Duration & period) override;

private:
  std::vector<std::string> joint_names_;

  // 阻抗参数
  std::vector<double> stiffness_;   // K (刚度)
  std::vector<double> damping_;     // D (阻尼)
  std::vector<double> mass_;        // M (惯性)

  // 期望轨迹
  std::vector<double> desired_position_;
  std::vector<double> desired_velocity_;

  // 外力
  std::vector<double> external_force_;

  // 订阅器
  rclcpp::Subscription<geometry_msgs::msg::WrenchStamped>::SharedPtr force_sub_;

  void force_callback(const geometry_msgs::msg::WrenchStamped::SharedPtr msg);
  double compute_torque(size_t joint_idx, double pos_error, double vel_error, double ext_force);
};

} // namespace openarm_impedance_controller

#endif
```

**src/impedance_controller.cpp:**
```cpp
#include "openarm_impedance_controller/impedance_controller.hpp"

namespace openarm_impedance_controller
{

controller_interface::CallbackReturn ImpedanceController::on_init()
{
  // 声明参数
  auto_declare<std::vector<std::string>>("joints", std::vector<std::string>());
  auto_declare<std::vector<double>>("stiffness", std::vector<double>(7, 100.0));
  auto_declare<std::vector<double>>("damping", std::vector<double>(7, 10.0));
  auto_declare<std::vector<double>>("mass", std::vector<double>(7, 1.0));

  return controller_interface::CallbackReturn::SUCCESS;
}

controller_interface::InterfaceConfiguration
ImpedanceController::command_interface_configuration() const
{
  controller_interface::InterfaceConfiguration config;
  config.type = controller_interface::interface_configuration_type::INDIVIDUAL;

  for (const auto & joint_name : joint_names_) {
    config.names.push_back(joint_name + "/effort");
  }

  return config;
}

controller_interface::InterfaceConfiguration
ImpedanceController::state_interface_configuration() const
{
  controller_interface::InterfaceConfiguration config;
  config.type = controller_interface::interface_configuration_type::INDIVIDUAL;

  for (const auto & joint_name : joint_names_) {
    config.names.push_back(joint_name + "/position");
    config.names.push_back(joint_name + "/velocity");
  }

  return config;
}

controller_interface::CallbackReturn ImpedanceController::on_configure(
  const rclcpp_lifecycle::State & /*previous_state*/)
{
  // 读取参数
  joint_names_ = get_node()->get_parameter("joints").as_string_array();
  stiffness_ = get_node()->get_parameter("stiffness").as_double_array();
  damping_ = get_node()->get_parameter("damping").as_double_array();
  mass_ = get_node()->get_parameter("mass").as_double_array();

  // 初始化
  size_t num_joints = joint_names_.size();
  desired_position_.resize(num_joints, 0.0);
  desired_velocity_.resize(num_joints, 0.0);
  external_force_.resize(num_joints, 0.0);

  // 订阅外力
  force_sub_ = get_node()->create_subscription<geometry_msgs::msg::WrenchStamped>(
    "~/force_sensor", 10,
    std::bind(&ImpedanceController::force_callback, this, std::placeholders::_1));

  RCLCPP_INFO(get_node()->get_logger(), "Impedance controller configured");
  return controller_interface::CallbackReturn::SUCCESS;
}

controller_interface::CallbackReturn ImpedanceController::on_activate(
  const rclcpp_lifecycle::State & /*previous_state*/)
{
  // 将当前位置设为期望位置
  for (size_t i = 0; i < joint_names_.size(); ++i) {
    desired_position_[i] = state_interfaces_[i * 2].get_value();  // position
  }

  return controller_interface::CallbackReturn::SUCCESS;
}

controller_interface::CallbackReturn ImpedanceController::on_deactivate(
  const rclcpp_lifecycle::State & /*previous_state*/)
{
  return controller_interface::CallbackReturn::SUCCESS;
}

controller_interface::return_type ImpedanceController::update(
  const rclcpp::Time & /*time*/, const rclcpp::Duration & /*period*/)
{
  // 阻抗控制律: τ = M(ẍd - ẍ) + D(ẋd - ẋ) + K(xd - x) + τext

  for (size_t i = 0; i < joint_names_.size(); ++i) {
    // 当前状态
    double current_position = state_interfaces_[i * 2].get_value();
    double current_velocity = state_interfaces_[i * 2 + 1].get_value();

    // 误差
    double pos_error = desired_position_[i] - current_position;
    double vel_error = desired_velocity_[i] - current_velocity;

    // 计算力矩
    double torque = compute_torque(i, pos_error, vel_error, external_force_[i]);

    // 输出力矩命令
    command_interfaces_[i].set_value(torque);
  }

  return controller_interface::return_type::OK;
}

void ImpedanceController::force_callback(const geometry_msgs::msg::WrenchStamped::SharedPtr msg)
{
  // 简化: 将笛卡尔力映射到关节空间
  // 实际应用需要雅可比转置
  external_force_[0] = msg->wrench.force.x;
  // ... (完整的力映射)
}

double ImpedanceController::compute_torque(
  size_t joint_idx, double pos_error, double vel_error, double ext_force)
{
  // τ = K*Δx + D*Δẋ + τext
  return stiffness_[joint_idx] * pos_error +
         damping_[joint_idx] * vel_error +
         ext_force;
}

} // namespace openarm_impedance_controller
```

**visual_servo_node.py:**
```python
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image
from geometry_msgs.msg import Twist
from cv_bridge import CvBridge
import cv2
import numpy as np

class VisualServoNode(Node):
    def __init__(self):
        super().__init__('visual_servo_node')

        # 订阅相机图像
        self.image_sub = self.create_subscription(
            Image, '/camera/color/image_raw',
            self.image_callback, 10)

        # 发布速度命令
        self.vel_pub = self.create_publisher(Twist, '/left_arm/cartesian_velocity', 10)

        self.bridge = CvBridge()

        # 目标特征 (例如: 红色物体中心)
        self.target_x = 320  # 图像中心
        self.target_y = 240

        # 控制增益
        self.kp_x = 0.001
        self.kp_y = 0.001

        self.get_logger().info('Visual servo node started')

    def image_callback(self, msg):
        """处理图像并计算控制命令"""
        # 转换为OpenCV格式
        cv_image = self.bridge.imgmsg_to_cv2(msg, desired_encoding='bgr8')

        # 检测目标 (红色物体)
        hsv = cv2.cvtColor(cv_image, cv2.COLOR_BGR2HSV)
        lower_red = np.array([0, 100, 100])
        upper_red = np.array([10, 255, 255])
        mask = cv2.inRange(hsv, lower_red, upper_red)

        # 计算质心
        moments = cv2.moments(mask)
        if moments['m00'] > 0:
            cx = int(moments['m10'] / moments['m00'])
            cy = int(moments['m01'] / moments['m00'])

            # 误差
            error_x = cx - self.target_x
            error_y = cy - self.target_y

            # 视觉伺服控制律
            vel = Twist()
            vel.linear.x = -self.kp_y * error_y  # 前后
            vel.linear.y = -self.kp_x * error_x  # 左右
            vel.linear.z = 0.0
            vel.angular.x = 0.0
            vel.angular.y = 0.0
            vel.angular.z = 0.0

            self.vel_pub.publish(vel)

            self.get_logger().info(
                f'Target at ({cx}, {cy}), Error: ({error_x}, {error_y})')
        else:
            # 未检测到目标,停止运动
            self.vel_pub.publish(Twist())

def main():
    rclpy.init()
    node = VisualServoNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()
```
