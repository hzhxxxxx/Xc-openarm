  <!-- 包含子模型 -->
  <xacro:include filename="$(find openarmx_description)/urdf/left_arm.xacro"/>
  <xacro:include filename="$(find openarmx_description)/urdf/right_arm.xacro"/>

    </visual>
  </link>

  <!-- 左臂实例 -->
  <xacro:left_arm parent="base_link" prefix="left_">
    <origin xyz="0 0.15 0.1" rpy="0 0 0"/>
  </xacro:left_arm>

  <!-- 右臂实例 -->
  <xacro:right_arm parent="base_link" prefix="right_">
    <origin xyz="0 -0.15 0.1" rpy="0 0 0"/>

**核心类结构:**

**include/openarmx_hardware/openarm_system.hpp:**
```cpp
#ifndef openarmx_hardware__OPENARM_SYSTEM_HPP_
#define openarmx_hardware__OPENARM_SYSTEM_HPP_

#include <hardware_interface/system_interface.hpp>
#include <rclcpp/rclcpp.hpp>
#include "robstride_can_interface.hpp"

namespace openarmx_hardware
{

class OpenArmSystem : public hardware_interface::SystemInterface
{
public:
  // 生命周期回调
  CallbackReturn on_init(const hardware_interface::HardwareInfo & info) override;
  CallbackReturn on_configure(const rclcpp_lifecycle::State & previous_state) override;
  CallbackReturn on_activate(const rclcpp_lifecycle::State & previous_state) override;
  CallbackReturn on_deactivate(const rclcpp_lifecycle::State & previous_state) override;

  // 硬件通信
  hardware_interface::return_type read(
    const rclcpp::Time & time, const rclcpp::Duration & period) override;
  hardware_interface::return_type write(
    const rclcpp::Time & time, const rclcpp::Duration & period) override;

  // 接口导出
  std::vector<hardware_interface::StateInterface> export_state_interfaces() override;
  std::vector<hardware_interface::CommandInterface> export_command_interfaces() override;

private:
  // CAN总线接口
  std::shared_ptr<RobstrideCAN> can0_;  // 左臂
  std::shared_ptr<RobstrideCAN> can1_;  // 右臂

  // 电机控制器
  std::vector<std::shared_ptr<MotorController>> left_motors_;
  std::vector<std::shared_ptr<MotorController>> right_motors_;

  // 状态缓存
  std::vector<double> hw_positions_;
  std::vector<double> hw_velocities_;
  std::vector<double> hw_efforts_;

  // 命令缓存
  std::vector<double> hw_commands_;
};

} // namespace openarmx_hardware

#endif
```

**CMakeLists.txt关键部分:**
```cmake
cmake_minimum_required(VERSION 3.8)
project(openarmx_hardware)

# 依赖
find_package(ament_cmake REQUIRED)
find_package(hardware_interface REQUIRED)
find_package(pluginlib REQUIRED)
find_package(rclcpp REQUIRED)
find_package(rclcpp_lifecycle REQUIRED)

# 编译硬件接口库
add_library(${PROJECT_NAME} SHARED
  src/openarm_system.cpp
  src/robstride_can_interface.cpp
  src/motor_controller.cpp
)

target_include_directories(${PROJECT_NAME} PUBLIC
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
  $<INSTALL_INTERFACE:include>
)

ament_target_dependencies(${PROJECT_NAME}
  hardware_interface
  pluginlib
  rclcpp
  rclcpp_lifecycle
)

# 注册为pluginlib插件
pluginlib_export_plugin_description_file(hardware_interface openarmx_hardware_plugin.xml)

# 安装
install(TARGETS ${PROJECT_NAME}
  ARCHIVE DESTINATION lib
  LIBRARY DESTINATION lib
  RUNTIME DESTINATION bin
)

install(DIRECTORY include/
  DESTINATION include
)

ament_package()
```

**主启动文件 launch/openarm.launch.py:**
```python
#!/usr/bin/env python3
import os
from launch import LaunchDescription
from launch.actions import DeclareLaunchArgument, IncludeLaunchDescription
from launch.substitutions import LaunchConfiguration, PathJoinSubstitution
from launch_ros.actions import Node
from launch_ros.substitutions import FindPackageShare
from ament_index_python.packages import get_package_share_directory

def generate_launch_description():
    # 路径
    pkg_description = FindPackageShare('openarmx_description')
    pkg_controllers = FindPackageShare('openarm_controllers')

    # 参数
    use_sim_time = LaunchConfiguration('use_sim_time', default='false')

    # URDF
    urdf_file = PathJoinSubstitution([
        pkg_description, 'urdf', 'openarm.urdf.xacro'
    ])

    # 控制器配置
    controllers_file = PathJoinSubstitution([
        pkg_controllers, 'config', 'openarm_controllers.yaml'
    ])

    # robot_state_publisher
    robot_state_publisher = Node(
        package='robot_state_publisher',
        executable='robot_state_publisher',
        parameters=[{
            'robot_description': Command(['xacro ', urdf_file]),
            'use_sim_time': use_sim_time
        }]
    )

    # controller_manager
    controller_manager = Node(
        package='controller_manager',
        executable='ros2_control_node',
        parameters=[controllers_file],
        output='both',
        remappings=[
            ('/controller_manager/robot_description', '/robot_description')
        ]
    )

    # spawner节点
    joint_state_broadcaster_spawner = Node(
        package='controller_manager',
        executable='spawner',
        arguments=['joint_state_broadcaster', '--controller-manager', '/controller_manager']
    )

    left_arm_controller_spawner = Node(
        package='controller_manager',
        executable='spawner',
        arguments=['left_arm_controller', '--controller-manager', '/controller_manager']
    )

    right_arm_controller_spawner = Node(
        package='controller_manager',
        executable='spawner',
        arguments=['right_arm_controller', '--controller-manager', '/controller_manager']
    )

    return LaunchDescription([
        DeclareLaunchArgument('use_sim_time', default_value='false'),
        robot_state_publisher,
        controller_manager,
        joint_state_broadcaster_spawner,
        left_arm_controller_spawner,
        right_arm_controller_spawner
    ])
```

**关键脚本示例 scripts/en_all_motors.py:**
```python
#!/usr/bin/env python3
"""
使能所有电机
"""
import can
import sys
import time
from lib.robstride import Motor
from lib.can_utils import check_can_interface

def main():
    # 检查CAN接口
    if not check_can_interface('can0') or not check_can_interface('can1'):
        print("❌ CAN接口未就绪,请先运行 en_all_can.py")
        sys.exit(1)

    # 连接CAN总线
    bus0 = can.interface.Bus(channel='can0', bustype='socketcan')
    bus1 = can.interface.Bus(channel='can1', bustype='socketcan')

    # 左臂电机 (ID 1-7, can0)
    print("使能左臂电机...")
    for motor_id in range(1, 8):
        try:
            motor = Motor(bus0, motor_id)
            motor.enable()
            print(f"  ✓ 左臂电机 {motor_id} 使能成功")
            time.sleep(0.1)
        except Exception as e:
            print(f"  ✗ 左臂电机 {motor_id} 使能失败: {e}")

    # 右臂电机 (ID 1-7, can1)
    print("\n使能右臂电机...")
    for motor_id in range(1, 8):
        try:
            motor = Motor(bus1, motor_id)
            motor.enable()
            print(f"  ✓ 右臂电机 {motor_id} 使能成功")
            time.sleep(0.1)
        except Exception as e:
            print(f"  ✗ 右臂电机 {motor_id} 使能失败: {e}")

    print("\n✅ 所有电机使能完成")

if __name__ == '__main__':
    main()
```

## 5. 编译系统

### 5.1 编译顺序

```bash
# 1. 先编译消息包
colcon build --packages-select openarm_msgs

# 2. 编译描述包
colcon build --packages-select openarmx_description

# 3. 编译硬件接口
colcon build --packages-select openarmx_hardware

# 4. 编译控制器配置
colcon build --packages-select openarm_controllers

# 5. 编译MoveIt配置
colcon build --packages-select openarm_moveit_config

# 6. 编译启动包
colcon build --packages-select openarmx_bringup

# 或一次性编译全部
colcon build
```

### 5.2 编译选项

```bash
# 调试版本
colcon build --cmake-args -DCMAKE_BUILD_TYPE=Debug

# Release版本(性能优化)
colcon build --cmake-args -DCMAKE_BUILD_TYPE=Release

# 并行编译
colcon build --parallel-workers 4

# 仅编译修改的包及其依赖
colcon build --packages-up-to openarmx_hardware

# 详细输出
colcon build --event-handlers console_direct+

# 符号链接安装(开发模式)
colcon build --symlink-install
```

```

    ↓
硬件参数 (openarmx_hardware/config/)
    ↓

```

### 6.2 参数覆盖机制

```bash
# 启动时覆盖参数
ros2 launch openarmx_bringup openarm.launch.py \

## 10. 总结

### 10.1 包结构要点

```
✓ openarmx_description - 机器人模型(URDF)
✓ openarmx_hardware - 硬件接口(C++)
✓ openarm_controllers - 控制器配置(YAML)

✓ openarm_moveit_config - MoveIt配置
✓ openarm_msgs - 自定义消息

### 10.2 开发入口点

**扩展硬件功能:**
- 修改 `openarmx_hardware/src/openarm_system.cpp`

**添加控制器:**
- 编辑 `openarm_controllers/config/openarm_controllers.yaml`

**修改机器人模型:**
- 编辑 `openarmx_description/urdf/*.xacro`

**添加自定义节点:**
- 在 `openarmx_bringup` 创建新包

**定义新消息:**
- 在 `openarm_msgs` 添加 `.msg` / `.srv` / `.action`
