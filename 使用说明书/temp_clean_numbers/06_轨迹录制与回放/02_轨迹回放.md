# 轨迹回放

## 录制轨迹的回放与执行

重现已录制的机器人运动轨迹。


##  轨迹回放概述

### 回放原理

**工作流程:**
```
1. 加载YAML轨迹文件
   ↓
2. 解析轨迹点(positions, time_from_start)
   ↓
3. 生成ros2_control轨迹消息
   ↓
4. 发送到控制器
   ↓
5. 控制器执行轨迹
   ↓
6. 机器人重现录制的运动
```

### 回放方式

OpenArmX提供两种回放方式:

**方式1: 使用playback节点**
```bash
ros2 run openarm_record playback_trajectory
```

**方式2: 使用Python脚本**
```python
自定义Python脚本加载和回放
```

##  Python脚本回放

### 基本回放脚本

```python
#!/usr/bin/env python3
"""
轨迹回放脚本
"""
import rclpy
from rclpy.node import Node
from rclpy.action import ActionClient
from control_msgs.action import FollowJointTrajectory
from trajectory_msgs.msg import JointTrajectory, JointTrajectoryPoint
import yaml
from builtin_interfaces.msg import Duration

class TrajectoryPlayback(Node):
    def __init__(self):
        super().__init__('trajectory_playback')

        # 创建Action Client
        self.left_arm_client = ActionClient(
            self,
            FollowJointTrajectory,
            '/left_arm_controller/follow_joint_trajectory'
        )

        self.right_arm_client = ActionClient(
            self,
            FollowJointTrajectory,
            '/right_arm_controller/follow_joint_trajectory'
        )

        self.get_logger().info("轨迹回放节点初始化")

    def load_trajectory(self, filename):
        """加载YAML轨迹文件"""
        with open(filename, 'r') as f:
            data = yaml.safe_load(f)

        self.get_logger().info(f"加载轨迹: {filename}")
        self.get_logger().info(f"轨迹点数: {len(data['points'])}")

        return data

    def playback_left_arm(self, trajectory_data, speed_scale=1.0):
        """回放左臂轨迹"""
        # 构建JointTrajectory消息
        traj_msg = JointTrajectory()

        # 设置关节名称(仅左臂)
        traj_msg.joint_names = [
            'left_joint_1', 'left_joint_2', 'left_joint_3',
            'left_joint_4', 'left_joint_5', 'left_joint_6',
            'left_joint_7'
        ]

        # 提取左臂轨迹点
        for point_data in trajectory_data['points']:
            point = JointTrajectoryPoint()

            # 提取左臂关节位置(前7个)
            point.positions = point_data['positions'][:7]

            # 时间缩放
            time_sec = point_data['time_from_start']['sec']
            time_nsec = point_data['time_from_start']['nanosec']
            total_sec = time_sec + time_nsec * 1e-9
            scaled_sec = total_sec / speed_scale

            point.time_from_start = Duration(
                sec=int(scaled_sec),
                nanosec=int((scaled_sec - int(scaled_sec)) * 1e9)
            )

            traj_msg.points.append(point)

        # 创建Goal
        goal_msg = FollowJointTrajectory.Goal()
        goal_msg.trajectory = traj_msg

        # 等待Action Server
        self.get_logger().info("等待左臂控制器...")
        self.left_arm_client.wait_for_server()

        # 发送Goal
        self.get_logger().info(f"发送左臂轨迹(速度缩放: {speed_scale})")
        future = self.left_arm_client.send_goal_async(goal_msg)

        return future

    def playback_right_arm(self, trajectory_data, speed_scale=1.0):
        """回放右臂轨迹"""
        traj_msg = JointTrajectory()

        traj_msg.joint_names = [
            'right_joint_1', 'right_joint_2', 'right_joint_3',
            'right_joint_4', 'right_joint_5', 'right_joint_6',
            'right_joint_7'
        ]

        for point_data in trajectory_data['points']:
            point = JointTrajectoryPoint()

            # 提取右臂关节位置(后7个,索引7-13)
            point.positions = point_data['positions'][7:14]

            time_sec = point_data['time_from_start']['sec']
            time_nsec = point_data['time_from_start']['nanosec']
            total_sec = time_sec + time_nsec * 1e-9
            scaled_sec = total_sec / speed_scale

            point.time_from_start = Duration(
                sec=int(scaled_sec),
                nanosec=int((scaled_sec - int(scaled_sec)) * 1e9)
            )

            traj_msg.points.append(point)

        goal_msg = FollowJointTrajectory.Goal()
        goal_msg.trajectory = traj_msg

        self.get_logger().info("等待右臂控制器...")
        self.right_arm_client.wait_for_server()

        self.get_logger().info(f"发送右臂轨迹(速度缩放: {speed_scale})")
        future = self.right_arm_client.send_goal_async(goal_msg)

        return future

def main():
    import sys

    if len(sys.argv) < 2:
        print("用法: python3 playback.py <轨迹文件.yaml> [速度缩放]")
        sys.exit(1)

    filename = sys.argv[1]
    speed_scale = float(sys.argv[2]) if len(sys.argv) > 2 else 1.0

    rclpy.init()

    playback = TrajectoryPlayback()

    # 加载轨迹
    trajectory_data = playback.load_trajectory(filename)

    # 选择回放组
    choice = input("回放: (1)左臂 (2)右臂 (3)双臂: ")

    if choice == '1':
        playback.playback_left_arm(trajectory_data, speed_scale)
    elif choice == '2':
        playback.playback_right_arm(trajectory_data, speed_scale)
    elif choice == '3':
        playback.playback_left_arm(trajectory_data, speed_scale)
        playback.playback_right_arm(trajectory_data, speed_scale)

    # 等待完成
    rclpy.spin_once(playback, timeout_sec=trajectory_data['points'][-1]['time_from_start']['sec'] + 5)

    playback.get_logger().info("✓ 回放完成")

    rclpy.shutdown()

if __name__ == '__main__':
    main()
```

**运行:**
```bash

```

def loop_playback(self, trajectory_data, speed_scale=1.0, num_loops=3):
    """
    循环回放轨迹
    """
    for i in range(num_loops):
        self.get_logger().info(f"第 {i+1}/{num_loops} 次回放")

        future = self.playback_left_arm(trajectory_data, speed_scale)

        # 等待完成
        rclpy.spin_until_future_complete(self, future)

        result = future.result()
        if result:
            self.get_logger().info(f"第 {i+1} 次完成")
        else:
            self.get_logger().error(f"第 {i+1} 次失败")
            break

        # 间隔
        import time
        time.sleep(2.0)

    self.get_logger().info("循环回放结束")

# 使用
playback.loop_playback(trajectory_data, speed_scale=0.5, num_loops=3)

```python
from rclpy.action import ActionClient
from control_msgs.action import FollowJointTrajectory

def playback_with_feedback(self, trajectory_data, speed_scale=1.0):
    """
    回放并显示实时反馈
    """
    # 构建轨迹消息(省略,同上)
    traj_msg = self.build_trajectory_message(trajectory_data, speed_scale)

    goal_msg = FollowJointTrajectory.Goal()
    goal_msg.trajectory = traj_msg

    # 发送Goal并注册回调
    send_goal_future = self.left_arm_client.send_goal_async(
        goal_msg,
        feedback_callback=self.feedback_callback
    )

    rclpy.spin_until_future_complete(self, send_goal_future)

    goal_handle = send_goal_future.result()

    if not goal_handle.accepted:
        self.get_logger().error("Goal被拒绝")
        return

    self.get_logger().info("Goal已接受,开始执行...")

    # 等待结果
    result_future = goal_handle.get_result_async()
    rclpy.spin_until_future_complete(self, result_future)

    result = result_future.result().result
    self.get_logger().info(f"回放完成,错误码: {result.error_code}")

def feedback_callback(self, feedback_msg):
    """反馈回调"""
    feedback = feedback_msg.feedback

    # 计算进度
    if len(feedback.desired.positions) > 0:
        # 显示当前位置
        self.get_logger().info(
            f"进度: desired={feedback.desired.positions[0]:.3f}, "
            f"actual={feedback.actual.positions[0]:.3f}"
        )
```

def playback_segment(self, trajectory_data, start_idx, end_idx, speed_scale=1.0):
    """
    回放轨迹片段
    start_idx, end_idx: 轨迹点索引
    """
    # 提取片段
    segment_points = trajectory_data['points'][start_idx:end_idx]

    # 重新调整时间戳(从0开始)
    first_time = segment_points[0]['time_from_start']['sec'] + \
                 segment_points[0]['time_from_start']['nanosec'] * 1e-9

    for point in segment_points:
        original_time = point['time_from_start']['sec'] + \
                       point['time_from_start']['nanosec'] * 1e-9

        adjusted_time = original_time - first_time

        point['time_from_start']['sec'] = int(adjusted_time)
        point['time_from_start']['nanosec'] = int((adjusted_time - int(adjusted_time)) * 1e9)

    # 构建片段轨迹
    segment_traj = {
        'joint_names': trajectory_data['joint_names'],
        'points': segment_points
    }

    # 回放
    return self.playback_left_arm(segment_traj, speed_scale)

def mirror_playback(self, left_trajectory_data, speed_scale=1.0):
    """
    左臂轨迹镜像到右臂回放
    """
    # 复制轨迹数据
    import copy
    right_traj = copy.deepcopy(left_trajectory_data)

    # 修改关节名称
    right_traj['joint_names'] = [
        'right_joint_1', 'right_joint_2', 'right_joint_3',
        'right_joint_4', 'right_joint_5', 'right_joint_6',
        'right_joint_7'
    ]

    # 镜像positions (joint_1取反)
    for point in right_traj['points']:
        positions = list(point['positions'][:7])  # 左臂7个关节
        positions[0] = -positions[0]  # joint_1反向
        point['positions'] = positions

    # 回放右臂
    return self.playback_right_arm(right_traj, speed_scale)

### 误差监控

```python
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import JointState
import numpy as np

class TrajectoryMonitor(Node):
    def __init__(self):
        super().__init__('traj_monitor')

        self.subscription = self.create_subscription(
            JointState,
            '/joint_states',
            self.joint_state_callback,
            10
        )

        self.desired_positions = None  # 从轨迹文件加载
        self.trajectory_index = 0
        self.errors = []

    def joint_state_callback(self, msg):
        """监控关节误差"""
        if self.desired_positions is None:
            return

        # 获取左臂关节实际位置
        left_joint_names = [f'left_joint_{i}' for i in range(1, 8)]
        actual_positions = []

        for name in left_joint_names:
            if name in msg.name:
                idx = msg.name.index(name)
                actual_positions.append(msg.position[idx])

        # 计算误差
        if len(actual_positions) == 7 and self.trajectory_index < len(self.desired_positions):
            desired = self.desired_positions[self.trajectory_index]
            error = np.array(actual_positions) - np.array(desired)
            rms_error = np.sqrt(np.mean(error**2))

            self.errors.append(rms_error)

            self.get_logger().info(f"RMS误差: {rms_error:.4f} rad")

            self.trajectory_index += 1

    def get_statistics(self):
        """获取误差统计"""
        if len(self.errors) > 0:
            print(f"平均误差: {np.mean(self.errors):.4f} rad")
            print(f"最大误差: {np.max(self.errors):.4f} rad")
            print(f"最小误差: {np.min(self.errors):.4f} rad")
            print(f"标准差: {np.std(self.errors):.4f} rad")
```

def concatenate_trajectories(traj1_data, traj2_data):
    """
    拼接两个轨迹
    """
    import copy

    merged = copy.deepcopy(traj1_data)

    # 获取traj1的结束时间
    last_time_1 = traj1_data['points'][-1]['time_from_start']['sec'] + \
                  traj1_data['points'][-1]['time_from_start']['nanosec'] * 1e-9

    # 将traj2的时间偏移
    for point in traj2_data['points']:
        adjusted_point = copy.deepcopy(point)

        original_time = point['time_from_start']['sec'] + \
                       point['time_from_start']['nanosec'] * 1e-9

        new_time = last_time_1 + original_time

        adjusted_point['time_from_start']['sec'] = int(new_time)
        adjusted_point['time_from_start']['nanosec'] = int((new_time - int(new_time)) * 1e9)

        merged['points'].append(adjusted_point)

    return merged

# 使用
traj1 = playback.load_trajectory('traj1.yaml')
traj2 = playback.load_trajectory('traj2.yaml')

merged_traj = concatenate_trajectories(traj1, traj2)

# 保存
with open('merged_traj.yaml', 'w') as f:
    yaml.dump(merged_traj, f)
